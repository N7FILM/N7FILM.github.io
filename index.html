import os import json import base64 import requests import cpi from bs4 import BeautifulSoup import gspread from oauth2client.service_account import ServiceAccountCredentials import google.generativeai as genai from datetime import datetime # ========================================== # CONFIGURATION & API KEYS # ========================================== TMDB_API_KEY = os.getenv("TMDB_API_KEY", "your_tmdb_key") YOUTUBE_API_KEY = os.getenv("YOUTUBE_API_KEY", "your_yt_key") GEMINI_API_KEY = os.getenv("GEMINI_API_KEY", "your_gemini_key") GOOGLE_MAPS_API_KEY = os.getenv("GOOGLE_MAPS_API_KEY", "your_maps_key") genai.configure(api_key=GEMINI_API_KEY) cpi.update() # Ensure CPI data is up-to-date for inflation calculations # Setup Google Sheets scope = ['https://spreadsheets.google.com/feeds', 'https://www.googleapis.com/auth/drive'] creds = ServiceAccountCredentials.from_json_keyfile_name('credentials.json', scope) client = gspread.authorize(creds) sheet = client.open('NextGen_Movie_Intelligence').sheet1 # ========================================== # MODULE 1: EXTREME DATA AGGREGATION # ========================================== def fetch_tmdb_base(movie_id): """Fetches core movie data and credits from TMDB.""" url = f"https://api.themoviedb.org/3/movie/{movie_id}?api_key={TMDB_API_KEY}&append_to_response=credits,keywords" resp = requests.get(url).json() return resp def fetch_youtube_analysis(title): """Hunts YouTube for niche theory and trivia analysis, sorting by viewCount.""" query = f"{title} movie theory OR trivia analysis" url = f"https://www.googleapis.com/youtube/v3/search?part=snippet&q={query}&type=video&order=viewCount&key={YOUTUBE_API_KEY}&maxResults=2" resp = requests.get(url).json() videos = if "items" in resp: for item in resp["items"]: videos.append(item['id']['videoId']) return videos def fetch_filming_coordinates(location_string): """Translates a location string to lat/lng using Google Maps Geocoding API.""" url = f"https://maps.googleapis.com/maps/api/geocode/json?address={location_string}&key={GOOGLE_MAPS_API_KEY}" resp = requests.get(url).json() if resp['status'] == 'OK': loc = resp['results']['geometry']['location'] return {"lat": loc['lat'], "lng": loc['lng']} return None def adjust_box_office_inflation(nominal_revenue, release_year): """Calculates true historical weight using the CPI library.""" try: real_value = cpi.inflate(nominal_revenue, release_year) return round(real_value, 2) except cpi.errors.CPIObjectDoesNotExist: return nominal_revenue # Fallback if year is out of bounds def scrape_shotonwhat_tech_specs(title): """Simulated scraping of ShotOnWhat for camera and VFX data.""" # In a production environment, this utilizes BeautifulSoup against the ShotOnWhat DOM. # For demonstration, returning a structured mock payload based on research. return { "camera": "ARRI ALEXA Mini, IMAX 3D-30", "lenses": "Panavision C Series Anamorphic", "vfx_software": "Autodesk After Effects, Nuke", "film_stock": "Digital 4K UHD" } # ========================================== # MODULE 2: LLM MASTER CRITIC SYNTHESIS # ========================================== def generate_master_critic_review(movie_data, tech_specs): """Utilizes Gemini to write a multi-dimensional, semantic film review.""" model = genai.GenerativeModel('gemini-pro') prompt = f""" Act as a master film critic writing for a prestigious cinematic journal. Write a profound, multi-dimensional review for the movie '{movie_data['title']}' ({movie_data['release_date'][:4]}). Do not merely summarize the plot. Analyze the cinematography based on these tools: {tech_specs['camera']} and {tech_specs['lenses']}. Discuss the thematic depth, pacing, and how the visual aesthetic (color palette) influences the emotional resonance. Compare it to its contemporaries in the {movie_data['genres']['name']} genre. Maintain an authoritative, academic, yet visceral tone. """ response = model.generate_content(prompt) return response.text # ========================================== # MODULE 3: FRONTEND CODE GENERATION # ========================================== def generate_single_file_html(aggregated_data): """Generates the massive, single-file HTML/CSS/JS payload.""" # Serialize data for injection into the JS environment json_data = json.dumps(aggregated_data) html_content = f""" Next-Gen Movie Intelligence Hub

Master Critic Analysis

Box Office Economics

Nominal Gross:

Inflation Adjusted (2024$):

Acquisition & VFX Hardware

    Cast & Crew Collaboration Topology

    Theory & Trivia Synthesis

    """ with open("dist_index.html", "w", encoding="utf-8") as file: file.write(html_content) print("Single-file deployment ready at dist_index.html") # ========================================== # MASTER ORCHESTRATION PIPELINE # ========================================== def run_genesis_pipeline(movie_id): print(f"Initiating pipeline for Movie ID: {movie_id}") # 1. Fetch Core Data tmdb_data = fetch_tmdb_base(movie_id) title = tmdb_data['title'] release_year = int(tmdb_data['release_date'][:4]) nominal_revenue = tmdb_data.get('revenue', 0) # 2. Extreme Data Aggregation adjusted_revenue = adjust_box_office_inflation(nominal_revenue, release_year) tech_specs = scrape_shotonwhat_tech_specs(title) yt_ids = fetch_youtube_analysis(title) # 3. Format Cast for Graphs (Top 15 actors) cast_data = [{"name": c["name"], "character": c["character"]} for c in tmdb_data['credits']['cast'][:15]] # 4. LLM Master Critic Analysis critic_review = generate_master_critic_review(tmdb_data, tech_specs) # 5. Compile Final Data Payload aggregated_data = { "title": title, "release_year": release_year, "revenue": nominal_revenue, "adjusted_revenue": adjusted_revenue, "tech_specs": tech_specs, "youtube_ids": yt_ids, "critic_review": critic_review, "cast": cast_data } # 6. Update Google Sheets Archival Backup row_data = [title, release_year, nominal_revenue, adjusted_revenue, tech_specs['camera'], critic_review[:500]] sheet.append_row(row_data) print("Google Sheet Successfully Updated.") # 7. Generate Final Frontend File generate_single_file_html(aggregated_data) if __name__ == "__main__": # Example execution using Interstellar's TMDB ID (157336) run_genesis_pipeline(157336)
